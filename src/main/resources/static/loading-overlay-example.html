How to Use in Your Pages

<!-- 1. Include the CSS -->
<link rel="stylesheet" href="/css/loading-overlay.css">

<!-- 2. Add the overlay HTML (before closing body tag) -->
<div id="loadingOverlay" class="loading-overlay">
    <div class="loading-content">
        <div class="loading-spinner"></div>
        <h4>Generating Your Results...</h4>
        <p>This may take up to 30 seconds.<br>Please don't close this window.</p>
    </div>
</div>

<!-- 3. Include the JavaScript -->
<script src="/js/loading-overlay.js"></script>

<!-- 4. Use the LoadingOverlay API -->
<script>
    // Automatically attach to all action buttons
    LoadingOverlay.attachToButtons('.action-btn');

    // Or attach to a specific button with custom message
    LoadingOverlay.attachToButton(
        document.getElementById('myButton'),
        'Processing...',
        'Please wait while we process your request.'
    );

    // Or manually control
    LoadingOverlay.show();  // Show with default message
    LoadingOverlay.hide();  // Hide the overlay
</script>
